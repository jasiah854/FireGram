var userInput = document.getElementById("user-name");
var userIdCounter = 1;
userInput.addEventListener("keydown", async(event) => {
  if (event.keyCode === 13) {
   event.preventDefault();
   document.getElementById("button2").click();
  }
});

async function addRow() {
  userIdCounter++
  const userId = userIdCounter; //userIdCounter will +=1 when btn is clicked
  const user = document.getElementById("user-name").value; // user's name  input
  var newTr = document.createElement("tr"); // create tr tag
  newTr.innerHTML = `<td> ${userId} </td> <td> ${user} </td> `; //give the tr content 
  document.getElementById("tbl").appendChild(newTr); //add tr to table
  document.getElementById("user-name").value = ""; // reset the value of html input to null
}
